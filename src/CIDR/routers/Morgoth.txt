! Morgoth Router Configuration (Cisco IOS)
! Connects Amroth to Switch3 (Balrog)

enable
configure terminal

hostname Morgoth

! Configure interfaces
interface FastEthernet0/0
 ip address 10.76.15.202 255.255.255.252
 no shutdown
 description Link to Amroth
exit

interface FastEthernet0/1
 ip address 10.76.4.1 255.255.254.0
 no shutdown
 description Link to Switch3 (Balrog)
exit

! Static routes
! Default route to Amroth
ip route 0.0.0.0 0.0.0.0 10.76.15.201

end
write memory
