! Amansul Router Configuration (Cisco IOS)
! Central router connecting to Eregion, Minastir, and Fornost

enable
configure terminal

! Set hostname
hostname Amansul

! Configure interfaces
interface FastEthernet0/0
 ip address 10.76.15.193 255.255.255.252
 no shutdown
 description Link to Eregion
exit

interface FastEthernet0/1
 ip address 10.76.15.213 255.255.255.252
 no shutdown
 description Link to Minastir
exit

interface FastEthernet1/0
 ip address 10.76.15.229 255.255.255.252
 no shutdown
 description Link to Fornost
exit

! Static routes using CIDR aggregation
! Route to Eregion side networks (B1, B3, B6, B8, C3, E2)
ip route 10.76.0.0 255.255.240.0 10.76.15.194

! Route to Minastir side networks (B5, B7, E3)
ip route 10.76.4.0 255.255.252.0 10.76.15.214
ip route 10.76.12.0 255.255.254.0 10.76.15.214

! Route to Fornost side networks (B4, C1)
ip route 10.76.8.0 255.255.252.0 10.76.15.230

! Save configuration
end
write memory
